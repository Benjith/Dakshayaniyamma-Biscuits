CREATE TABLE `eklavya`.`tbl_modules` ( `Module_Id` INT(11) NOT NULL AUTO_INCREMENT ,  `Name` VARCHAR(100) NOT NULL ,  `Superuser` INT(11) NOT NULL COMMENT '0=false 1=true ' ,  `Admin` INT(11) NOT NULL COMMENT '0=false 1=true ' ,  `employee` INT(11) NOT NULL COMMENT '0=false 1=true ' ,  `Submodule` INT(11) NOT NULL COMMENT '0=false 1=true ' ,    PRIMARY KEY  (`Module_Id`)) ENGINE = InnoDB;
